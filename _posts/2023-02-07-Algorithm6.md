from collections import deque
t = int(input())
dx = [-1,1,0,0]
dy = [0,0,-1,1]

for _ in range(t):
    m,n,k = map(int,input().split())
    graph =[[0]*(m) for _ in range(n)]
    cnt = 0
    for i in range(k):
        x,y = map(int,input().split())
        graph[y][x] = 1
    for a in range(n):
        for b in range(m):
            if graph[a][b]==1:
                bfs(a,b)
                cnt+=1
    print(cnt)
