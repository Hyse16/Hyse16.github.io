# 파이썬(BFS사용)
```python
from collections import deque
n = int(input())
v = int(input())
graph = [[0]*(n+1) for _ in range(n+1)]
vis = [0]*(n+1)
for i in range(v):
    a,b = map(int,input().split())
    graph[a][b] = graph[b][a]= 1

def bfs(m):
    vis[m]=1
    
    q = deque()
    q.append(m)
    while q:
        node = q.popleft()
        for i in range(n+1):
            if graph[i][node] ==1 and vis[i]==0:
                q.append(i)
                vis[i] = 1
bfs(1)
print(sum(vis)-1)


```
